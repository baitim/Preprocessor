CALL MAIN:
HLT

MAIN:
    CALL PRINT_HELLO:

    PUSH 0
    POP  rax            / start value
    PUSH 7
    POP  rbx            / upper bound of values
    PUSH 1
    POP  rcx            / fact
    CALL FACT:
    RET

/ WARNING function changes registers values (rax rcx)
FACT:
    PUSH rax 
    PUSH rbx
    JAE  ret_fact:      / return
    CALL ADD_1_TO_RAX:
    CALL MUL_RAX_RCX:
    PUSH rcx
    OUT
    CALL FACT:          / calc and print again
    ret_fact:
    RET

ADD_1_TO_RAX:
    PUSH rax
    PUSH 1
    ADD
    POP  rax
    RET


MUL_RAX_RCX:
    PUSH rcx
    PUSH rax
    MUL
    POP  rcx
    RET

PRINT_HELLO:
    PUSH 72
    OUTC
    PUSH 69
    OUTC
    PUSH 76
    OUTC
    PUSH 76
    OUTC
    PUSH 79
    OUTC
    PUSH 33
    OUTC
    PUSH 10
    OUTC
    RET
